# DFA Configuration: Binary Odd Validator
# Description: Validates whether a binary string represents an odd number.
# Logic: Accepts any string ending in the digit '1'.

# Alphabet consists of binary digits 0 and 1.
ALPHABET: 0-1

# State definitions
# q0: Represents strings ending in '0' (Even) or the initial empty state.
# q1: Represents strings ending in '1' (Odd).
STATES: q0 q1

INITIAL_STATE: q0

# Strings ending in q1 are accepted as odd numbers.
FINAL_STATES: q1

TRANSITIONS:

# Transitions from q0 (Even/Start)
# Input '0' maintains the even state.
# Input '1' switches to the odd state.
q0 0 q0
q0 1 q1

# Transitions from q1 (Odd)
# Input '0' switches to the even state.
# Input '1' maintains the odd state.
q1 0 q0
q1 1 q1
