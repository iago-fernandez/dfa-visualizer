# DFA Configuration: Binary Odd Validator
# Description: Validates whether a binary string represents an odd number.
# Logic: Accepts any string ending in the digit '1'.

ALPHABET: 0-1

STATES: q_even q_odd

INITIAL_STATE: q_even
FINAL_STATES: q_odd

TRANSITIONS:

# State q_even (Start)
# Represents having read an even number (ending in 0) or empty string.
q_even 0 q_even
q_even 1 q_odd

# State q_odd
# Represents having read an odd number (ending in 1).
q_odd 0 q_even
q_odd 1 q_odd
